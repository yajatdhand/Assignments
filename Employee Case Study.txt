package com.atcs.Practice;

public class Employee {

	private int id;
	private String name;
	private int salary;
	private String address;
	private int phno;
	
	public Employee(String name, int salary, String address, int phno)
	{
		//this.id = id;
		this.name = name;
		this.salary = salary;
		this.address = address;
		this.phno = phno;
	}

	@Override
	public String toString() {
		return "Employee [name=" + name + ", salary=" + salary + ", address=" + address + ", phno=" + phno + "]";
	}

	
	
	

}
